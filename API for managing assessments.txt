info:
  description: API for managing assessments.
  version: 1.0.0
  title: Assessment MicroService
servers:
  - url: http://localhost:8080
paths:
  /assessments:
    get:
      tags: [assessment]
      summary: Get a list of assessments
      responses:
        '200':
          description: A list of assessments
          content:
            application/json:
              schema:
                type: array
                items:
                  type: object
                  properties:
                    sino:
                      type: string
                    setName:
                      type: string
                    createdBy:
                      type: string
                    domain:
                      type: string
                    status:
                      type: string
  /assessment/createset:
    post:
      tags: [assessment]
      summary: Create a new assessment
      requestBody:
        required: true
        content:
          application/json:
            schema:
              type: object
              properties:
                setName:
                  type: string
                domain:
                  type: string
                createdBy:
                  type: string
                createdAt:
                  type: string
                  format: date-time
                modifiedAt:
                  type: string
                  format: date-time
                questions:
                  type: array
                  items:
                    type: object
                    properties:
                      questionId:
                        type: string
                      questionText:
                        type: string
                      answers:
                        type: array
                        items:
                          type: object
                          properties:
                            answerId:
                              type: string
                            answer:
                              type: string
                            suggestion:
                              type: string
      responses:
        '201':
          description: Assessment created successfully
  /assessment/{setName}:
    get:
      tags: [assessment]
      summary: Get an assessment by setName
      parameters:
        - name: setName
          in: path
          required: true
          schema:
            type: string
      responses:
        '200':
          description: An assessment
          content:
            application/json:
              schema:
                type: object
                properties:
                  setName:
                    type: string
                  domain:
                    type: string
                  createdBy:
                    type: string
                  createdAt:
                    type: string
                    format: date-time
                  modifiedAt:
                    type: string
                    format: date-time
                  questions:
                    type: array
                    items:
                      type: object
                      properties:
                        questionId:
                          type: string
                        questionText:
                          type: string
                        answers:
                          type: array
                          items:
                            type: object
                            properties:
                              answerId:
                                type: string
                              answer:
                                type: string
                              suggestion:
                                type: string
  /assessment/{setName}/{questionId}:
    put:
      tags: [assessment]
      summary: Update a question in an assessment
      parameters:
        - name: setName
          in: path
          required: true
          schema:
            type: string
        - name: questionId
          in: path
          required: true
          schema:
            type: string
      requestBody:
        required: true
        content:
          application/json:
            schema:
              type: object
              properties:
                questionText:
                  type: string
                answers:
                  type: array
                  items:
                    type: object
                    properties:
                      answerId:
                        type: string
                      answer:
                        type: string
                      suggestion:
                        type: string
      responses:
        '200':
          description: Question updated successfully
    delete:
      tags: [assessment]
      summary: Delete a question from an assessment
      parameters:
        - name: setName
          in: path
          required: true
          schema:
            type: string
        - name: questionId
          in: path
          required: true
          schema:
            type: string
      responses:
        '204':
          description: Question deleted successfully
