  title: Survey API
  description: API documentation for the Survey application.
  version: 1.0.0
servers:
  - url: http://localhost:8080/
paths:
  /dashboard:
    get:
      summary: Dashboard
      description: Retrieves the dashboard.
      responses:
        '200':
          description: Successfully retrieved the dashboard.

 

  /createassessment:
    get:
      summary: Get All Assessments
      description: Retrieves a list of all assessments.
      responses:
        '200':
          description: Successfully retrieved the list of assessments.
          content:
            application/json:
              schema:
                type: array
                items:
                  type: object
                  properties:
                    slno:
                      type: string
                    setname:
                      type: string
                    createdby:
                      type: string
                    domain:
                      type: string
                    status:
                      type: string

 
  /createassessment/newassessment:
    post:
      summary: Create New Assessment
      description: Creates a new assessment with the provided details.
      requestBody:
        content:
          application/json:
            schema:
              type: object
              properties:
                setname:
                  type: string
                domain:
                  type: string
              required:
                - setname
                - domain
      responses:
        '201':
          description: Successfully created the new assessment.

 

  /createassessment/{setname}/addquestion:
    post:
      summary: Add Question to Assessment
      description: Adds a question to the specified assessment set.
      parameters:
        - name: setname
          in: path
          required: true
          schema:
            type: string
      requestBody:
        content:
          application/json:
            schema:
              type: object
              properties:
                setname:
                  type: string
                domain:
                  type: string
                createdby:
                  type: string
                createdtime:
                  type: string
                  format: date-time
                questions:
                  type: array
                  items:
                    type: object
                    properties:
                      question:
                        type: string
                      options:
                        type: array
                        items:
                          type: object
                          properties:
                            answerId:
                              type: string
                            answer:
                              type: string
                            suggestion:
                              type: string
      responses:
        '201':
          description: Successfully added the question to the assessment.

 

  /createassessment/{setname}/question/{questionid}:
    put:
      summary: Update Question in Assessment
      description: Updates the details of a specific question in the assessment set.
      parameters:
        - name: setname
          in: path
          required: true
          schema:
            type: string
        - name: questionid
          in: path
          required: true
          schema:
            type: string
      requestBody:
        content:
          application/json:
            schema:
              type: object
              properties:
                question:
                  type: string
                options:
                  type: array
                  items:
                    type: object
                    properties:
                      answerId:
                        type: string
                      answer:
                        type: string
                      suggestion:
                        type: string
      responses:
        '200':
          description: Successfully updated the question.

 

    delete:
      summary: Delete Question from Assessment
      description: Deletes a specific question from the assessment set.
      parameters:
        - name: setname
          in: path
          required: true
          schema:
            type: string
        - name: questionid
          in: path
          required: true
          schema:
            type: string
      responses:
        '204':
          description: Successfully deleted the question.

 

  /createassessment/{setname}/questions:
    get:
      summary: Get Questions from Assessment
      description: Retrieves a list of questions from a specific assessment set.
      parameters:
        - name: setname
          in: path
          required: true
          schema:
            type: string
      responses:
        '200':
          description: Successfully retrieved the list of questions.
          content:
            application/json:
              schema:
                type: array
                items:
                  type: object
                  properties:
                    questionId:
                      type: string
                    question:
                      type: string
                    options:
                      type: array
                      items:
                        type: object
                        properties:
                          answerId:
                            type: string
                          answer:
                            type: string
                          suggestion:
                            type: string

 

  /createassessment/{setname}/questions/{questionid}/answers:
    post:
      summary: Add Answers to Question
      description: Adds answers to a specific question in the assessment set.
      parameters:
        - name: setname
          in: path
          required: true
          schema:
            type: string
        - name: questionid
          in: path
          required: true
          schema:
            type: string
      requestBody:
        content:
          application/json:
            schema:
              type: array
              items:
                type: object
                properties:
                  answerId:
                    type: string
                  answer:
                    type: string
                  suggestion:
                    type: string
      responses:
        '201':
          description: Successfully added answers to the question.

 

  /createassessment/{newsetname}/fromexisting/{existingsetname}/questions/{questionid}:
    post:
      summary: Add Question from Existing Set to New Set
      description: Adds a question from an existing assessment set to a new assessment set.
      parameters:
        - name: newsetname
          in: path
          required: true
          schema:
            type: string
        - name: existingsetname
          in: path
          required: true
          schema:
            type: string
        - name: questionid
          in: path
          required: true
          schema:
            type: string
      responses:
        '201':
          description: Successfully added the question from the existing set to the new set.

      requestBody:
        content:
          application/json:
            schema:
              type: object

                domain:
                  type: string
                email:
                  type: string
                id:
                  type: string
                setName:
                  type: string
                status:
                  type: string
                companyName:
                  type: string
              required:
                - domain
                - email
                - id
                - setName
                - status
                - companyName
      responses:
        '201':
          description: Successfully created the new survey.
